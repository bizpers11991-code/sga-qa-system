export type Role =
  | 'asphalt_foreman'

  | 'asphalt_engineer'

  | 'spray_admin'

  | 'tender_admin'
  | 'management_admin'

  return [
    'asphalt_engineer', 'profiling_engineer', 'scheduler_admin',
    'management_admin', 'hseq_manager', 'spray_admin', 'tender_admin'
  ].includes(role);
};

  // Project Details
  projectName: string;

  location: string;
  estimatedStartDate: string;
  estimatedEndDate: string;

    asphalt: boolean;
    profiling: boolean;
    spray: boolean;

  // Status
  status: 'Draft' | 'Submitted' | 'Active' | 'Completed' | 'On Hold';
}
